/*

CREATE VIEW dbo.[Совершеннолетние проживающие]
AS 
	SELECT 
		[Клиенты].[Код клиента], 
		[Клиенты].[Код состояния], 
		[Клиенты].[Дата рождения], 
		[Клиенты].[Фамилия], 
		[Клиенты].[Имя], 
		[Клиенты].[Отчество], 
		[Категория номеров].[Категория], 
		[Категория номеров].[Стоимость в сутки]
	FROM ([Состояние номера] 
		INNER JOIN [Клиенты] 
		ON [Состояние номера].[Код номера] = [Клиенты].[Код состояния]) 
		INNER JOIN ([Категория номеров] 
		INNER JOIN [Номера] 
		ON [Категория номеров].[Код категории] = [Номера].[Категория]) 
		ON [Состояние номера].[Код номера] = [Номера].[Код номера]
	WHERE ((([Клиенты].[Дата рождения]) < CONVERT(DATETIME, '2004-01-01T00:00:00.000', 126)))
GO
